.PHONY: help install run-api run-bot run-all docker-build docker-up docker-down clean

help:
	@echo "Available commands:"
	@echo "  make install     - Install dependencies"
	@echo "  make run-api     - Run API server"
	@echo "  make run-bot     - Run Telegram bot"
	@echo "  make run-all     - Run both API and bot"
	@echo "  make docker-build- Build Docker images"
	@echo "  make docker-up   - Start all services with Docker"
	@echo "  make docker-down - Stop all services"
	@echo "  make clean       - Clean cache and logs"

install:
	pip install -r requirements.txt

run-api:
	python api.py

run-bot:
	python bot.py

run-all:
	@echo "Starting API server..."
	@python api.py &
	@sleep 2
	@echo "Starting Bot..."
	@python bot.py

docker-build:
	docker-compose build

docker-up:
	docker-compose up -d

docker-down:
	docker-compose down

docker-prod:
	docker-compose -f docker-compose.prod.yml up -d

clean:
	find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	find . -type f -name "*.pyc" -delete
	rm -rf logs/*.log
	rm -rf data/media/*
	rm -rf data/checks/*
	@echo "Cleaned cache and logs"
